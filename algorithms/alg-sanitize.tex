\begin{algorithm}[H]
  \caption{\label{alg.sanitize}
    The \emph{sanitize} function ran by party $\RB[j]$
    at round $r$ on an underlying ledger $L = \wheel[i][j][r]$
    sanitizes the underlying ledger by decoding and deserializing
    each of its transactions into a sequence of messages.
  }
  \begin{algorithmic}[1]
    \Function{\sf sanitize}{$j, i, L$}
      \Let{\textsf{ret}}{[\,]}
      \For{$\tx \in L$}
        \Let{m}{\textsf{decodeRollerbladeTx}(j, \textsf{tx})}
        \State{$\textsf{ret.append}(m)$}
      \EndFor
      \State\Return$\textsf{ret}$
    \EndFunction
    \vskip2pt
  \end{algorithmic}
\end{algorithm}
