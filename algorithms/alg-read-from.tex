\begin{algorithm}[H]
  \caption{\label{alg.read-from}
    The \emulateMachine function
    made available to the external user by a \rollerblade party.
  }
  \begin{algorithmic}[1]
    \Function{\emulateMachine}{$i, \simulationRound$}
      \Let{\Ledger[j][i]}{\this.\Y[j][i].\lread()}
      \Let{Z}{\this.\simulate(i, \Ledger[j][i], \simulationRound).Z}
      \State\Return{$Z$}
    \EndFunction
    \vskip2pt
  \end{algorithmic}
\end{algorithm}
