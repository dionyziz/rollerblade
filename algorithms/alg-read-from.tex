\begin{algorithm}[H]
  \caption{\label{alg.read-from}
    The \emulateMachine function
    made available to the external user by a \rollerblade party.
  }
  \begin{algorithmic}[1]
    \Function{\emulateMachine}{$\sid, \Pi, i, \Y[j], \simulationRound, \realityRound$}
      \Let{\Ledger[j][i]}{\Y[j][i].\lread()}
      \Let{Z}{\simulate(\sid, \Pi, i, \Ledger[j][i], \Y[j], \simulationRound, \realityRound).Z}
      \State\Return{$Z$}
    \EndFunction
    \vskip2pt
  \end{algorithmic}
\end{algorithm}
