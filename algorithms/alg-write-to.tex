\begin{algorithm}[H]
  \caption{\label{alg.write-to}
    The \emph{writeToMachine} function
    made available to the external user by a \rollerblade party.
  }
  \begin{algorithmic}[1]
    \Function{\writeToMachine}{$\sid, i, \data, \Y[j]$}
      \Let{\instruction}{\{\sid{:} \sid, \type{:} \text{`write'}, \data{:} \data\}}
      \Let{s}{\serialize(\instruction)}
      \Let{\tx}{\Y[j][i].\encode(s)}
      \State{$\Y[j][i].\lwrite(\tx)$}

    \EndFunction
    \vskip2pt
  \end{algorithmic}
\end{algorithm}
